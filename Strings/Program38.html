38. Program to replace lower-case characters with upper-case and vice-versa

Code


public class SwapCase {
    public static void main(String[] args) {
        String inputString = "Hello, World!";

        // Calling the method to swap case
        String swappedString = swapCase(inputString);

        // Displaying the result
        System.out.println("Input String: " + inputString);
        System.out.println("Swapped Case String: " + swappedString);
    }

    public static String swapCase(String str) {
        // Using a StringBuilder to efficiently build the swapped string
        StringBuilder swappedStringBuilder = new StringBuilder();

        // Iterating through each character in the string
        for (char c : str.toCharArray()) {
            // Checking if the character is uppercase
            if (Character.isUpperCase(c)) {
                // Converting uppercase to lowercase and appending to the StringBuilder
                swappedStringBuilder.append(Character.toLowerCase(c));
            } else if (Character.isLowerCase(c)) {
                // Converting lowercase to uppercase and appending to the StringBuilder
                swappedStringBuilder.append(Character.toUpperCase(c));
            } else {
                // Appending non-alphabetic characters directly to the StringBuilder
                swappedStringBuilder.append(c);
            }
        }

        // Converting the StringBuilder to a String and returning the result
        return swappedStringBuilder.toString();
    }
}

Output

Input String: Hello, World!
Swapped Case String: hELLO, wORLD!
